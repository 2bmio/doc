package main

import (
	"fmt"
	"time"
)

func main() {
	dayBorn := time.Monday
	switch dayBorn {
	case time.Monday:
		fmt.Println("on Monday")
	case time.Tuesday:
		fmt.Println("on Tuesday")
	case time.Wednesday:
		fmt.Println("on Wednesday")
	case time.Thursday:
		fmt.Println("on Thursday")
	case time.Friday:
		fmt.Println("on Friday")
	case time.Saturday:
		fmt.Println("on Saturday")
	case time.Sunday:
		fmt.Println("on Sunday")
	default:
		fmt.Println("Error, you not born yet!")
	}
}
